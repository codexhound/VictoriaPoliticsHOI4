on_government_change = {
	effect = {			
		if = {
			limit = {
				NOT = {
					OR = {
						has_government = fascism
						has_government = communism
						has_government = neutrality
					}
				}
			}
			country_event = {id = vc_elections.0 days = 730 random = 10}
		}
	}
}
